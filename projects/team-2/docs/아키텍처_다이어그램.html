flowchart TD
    Start(["진료 녹음 파일 수신"])
    --> STT["STT 실행 및 텍스트 변환"]
    --> Extract["진단/처방 및 관련 정보 추출"]
    --> D1{"진단/질환\n식별 가능?"}

    D1 -- "아니오" --> Check["기존 질환 여부 확인"]
    D1 -- "예" --> Check

    Check -- "아니오" --> NewThread["새 질환 관리\nthread 생성"]
    Check -- "예" --> ExistThread["기존 질환 관리\nthread 조회"]

    NewThread --> Merge["thread 정보 통합"]
    ExistThread --> Merge

    Merge --> D2{"의사 지침 &\n지식 추가 필요한가?"}

    D2 -- "아니오" --> Checklist["생활습관 체크리스트\n생성/업데이트"]
    D2 -- "필요시" --> RAG["보조 RAG/검색 기반\n지식 수집"]
    RAG --> SafeFilter["신뢰성 검증 &\n안전 필터링"]
    SafeFilter --> Checklist

    Checklist --> Verify["의료 안전성 검증 결과 반영"]

    Verify --> D3{"의료 전문가/환자 확인\n필요 여부 판단"}

    D3 -- "예" --> Save["환자 및 기록 저장"]
    Save --> End1(["종료"])

    D3 -- "thread 저장" --> ThreadSave["thread 상태 저장"]
    ThreadSave --> End1

    D3 -- "아니오" --> ThreadClose["thread 종료"]
    ThreadClose --> End2(["다른 질문"])

    %% 피드백 루프 (원본 점선 화살표)
    End2 -.->|"재처리"| Start
    Checklist -.->|"재분석"| Extract
